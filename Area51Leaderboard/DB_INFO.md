create table public.leaderboard (
  id bigint generated by default as identity not null,
  board text not null,
  name text not null,
  score numeric not null,
  created_at timestamp with time zone null default now(),
  venue text null,
  status smallint not null default 1,
  updated_at timestamp with time zone not null default now(),
  constraint leaderboard_pkey primary key (id),
  constraint leaderboard_board_check check (
    (
      board = any (
        array[
          'today'::text,
          'hist'::text,
          'both'::text,
          'girl'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_leaderboard_board_venue_status on public.leaderboard using btree (board, venue, status) TABLESPACE pg_default;

create index IF not exists idx_leaderboard_status_score on public.leaderboard using btree (status, score) TABLESPACE pg_default;

create index IF not exists idx_leaderboard_venue_board_score on public.leaderboard using btree (venue, board, score) TABLESPACE pg_default;
